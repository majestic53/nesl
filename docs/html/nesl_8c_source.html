34int nesl(const nesl_t *context)
35{
36 int result;
37
38 if((result = nesl_service_initialize(context->title, context->fullscreen, context->linear, context->scale)) == NESL_FAILURE) {
39 goto exit;
40 }
41
42 if((result = nesl_bus_initialize(context->data, context->length)) == NESL_FAILURE) {
43 goto exit;
44 }
45
46 while((result = nesl_service_poll()) == NESL_SUCCESS) {
47
48 while(!nesl_bus_cycle());
49
50 if((result = nesl_service_redraw()) == NESL_FAILURE) {
51 goto exit;
52 }
53 }
54
55exit:
56 nesl_bus_uninitialize();
57 nesl_service_uninitialize();
58
59 return result;
60}
