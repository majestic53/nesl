39static nesl_error_e nesl_cartridge_validate(const void *data, int length)
40{
41 nesl_error_e result = NESL_SUCCESS;
42 int expected = sizeof(nesl_cartridge_header_t);
43 const nesl_cartridge_header_t *header = (const nesl_cartridge_header_t *)data;
44
45 if(!data) {
46 result = SET_ERROR("Invalid data -- %p", data);
47 goto exit;
48 }
49
50 if(length < expected) {
51 result = SET_ERROR("Invalid length -- %.02f KB (%i bytes), expecting %.02f KB (%i bytes)", length / (float)1024, length, expected / (float)1024, expected);
52 goto exit;
53 }
54
55 if(strncmp(header->magic, "NES\x1A", 4)) {
56 result = SET_ERROR("Malformed header -- %s", "String mismatch");
57 goto exit;
58 }
59
60 if(header->flag_7.version) {
61 result = SET_ERROR("Unsupported header version -- %u", header->flag_7.version);
62 goto exit;
63 }
64
65 if(header->flag_7.vs_unisystem || header->flag_7.playchoice_10) {
66 result = SET_ERROR("Unsupported header type -- %s", header->flag_7.vs_unisystem ? "VS-Unisystem" : "PlayChoice-10");
67 goto exit;
68 }
69
70 if(header->flag_6.trainer) {
71 expected += 512;
72 }
73
74 expected += (header->rom.program * 16 * 1024) + (header->rom.character * 8 * 1024);
75
76 if(length != expected) {
77 result = SET_ERROR("Invalid length -- %.02f KB (%i bytes), expecting %.02f KB (%i bytes)", length / (float)1024, length, expected / (float)1024, expected);
78 goto exit;
79 }
80
81exit:
82 return result;
83}

116nesl_error_e nesl_cartridge_initialize(nesl_cartridge_t *cartridge, const void *data, int length)
117{
118 nesl_error_e result;
119 const uint8_t *offset = data;
120
121 if((result = nesl_cartridge_validate(data, length)) == NESL_FAILURE) {
122 goto exit;
123 }
124
125 cartridge->header = (const nesl_cartridge_header_t *)offset;
126 offset += sizeof(*cartridge->header);
127
128 if(cartridge->header->flag_6.trainer) {
129 offset += 512;
130 }
131
132 if(cartridge->header->rom.program) {
133 cartridge->rom.program = offset;
134 offset += (cartridge->header->rom.program * 16 * 1024);
135 }
136
137 if(cartridge->header->rom.character) {
138 cartridge->rom.character = offset;
139 offset += (cartridge->header->rom.character * 8 * 1024 );
140 } else {
141
142 if(!(cartridge->ram.character = calloc(32 * 8 * 1024, sizeof(uint8_t)))) {
143 result = SET_ERROR("Failed to allocate buffer -- %u KB (%i bytes)", 8, 8 * 1024);
144 goto exit;
145 }
146
147 cartridge->rom.character = cartridge->ram.character;
148 }
149
150 if(!(cartridge->ram.program = calloc((cartridge->header->ram.program ? cartridge->header->ram.program : 1) * 8 * 1024, sizeof(uint8_t)))) {
151 result = SET_ERROR("Failed to allocate buffer -- %u KB (%i bytes)", (cartridge->header->ram.program ? cartridge->header->ram.program : 1) * 8,
152 (cartridge->header->ram.program ? cartridge->header->ram.program : 1) * 8 * 1024);
153 goto exit;
154 }
155
156 memset(cartridge->ram.program, 0, (cartridge->header->ram.program ? cartridge->header->ram.program : 1) * 8 * 1024);
157
158exit:
159 return result;
160}
