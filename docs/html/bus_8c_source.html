59static nesl_error_e nesl_bus_reset(void)
60{
61 nesl_error_e result;
62
63 if((result = nesl_service_reset()) == NESL_FAILURE) {
64 goto exit;
65 }
66
67 if((result = nesl_mapper_reset(&g_bus.subsystem.mapper)) == NESL_FAILURE) {
68 goto exit;
69 }
70
71 if((result = nesl_audio_reset(&g_bus.subsystem.audio)) == NESL_FAILURE) {
72 goto exit;
73 }
74
75 if((result = nesl_input_reset(&g_bus.subsystem.input)) == NESL_FAILURE) {
76 goto exit;
77 }
78
79 if((result = nesl_processor_reset(&g_bus.subsystem.processor)) == NESL_FAILURE) {
80 goto exit;
81 }
82
83 if((result = nesl_video_reset(&g_bus.subsystem.video, &g_bus.subsystem.mapper.mirror)) == NESL_FAILURE) {
84 goto exit;
85 }
86
87 g_bus.cycle = 0;
88
89exit:
90 return result;
91}

93bool nesl_bus_cycle(void)
94{
95 nesl_processor_cycle(&g_bus.subsystem.processor, g_bus.cycle);
96 nesl_audio_cycle(&g_bus.subsystem.audio, g_bus.cycle);
97 ++g_bus.cycle;
98
99 return nesl_video_cycle(&g_bus.subsystem.video);
100}

130nesl_error_e nesl_bus_interrupt(nesl_interrupt_e type)
131{
132 int result = NESL_FAILURE;
133
134 switch(type) {
135 case INTERRUPT_MASKABLE:
136 case INTERRUPT_NON_MASKABLE:
137
138 if((result = nesl_processor_interrupt(&g_bus.subsystem.processor, type == INTERRUPT_MASKABLE)) == NESL_FAILURE) {
139 goto exit;
140 }
141 break;
142 case INTERRUPT_RESET:
143
144 if((result = nesl_bus_reset()) == NESL_FAILURE) {
145 goto exit;
146 }
147 break;
148 case INTERRUPT_MAPPER:
149
150 if((result = nesl_mapper_interrupt(&g_bus.subsystem.mapper)) == NESL_FAILURE) {
151 goto exit;
152 }
153 break;
154 default:
155 break;
156 }
157
158exit:
159 return result;
160}

162uint8_t nesl_bus_read(nesl_bus_e type, uint16_t address)
163{
164 uint8_t result = 0;
165
166 switch(type) {
167 case BUS_PROCESSOR:
168
169 switch(address) {
170 case 0x0000 ... 0x1FFF:
171 result = nesl_processor_read(&g_bus.subsystem.processor, address);
172 break;
173 case 0x2000 ... 0x3FFF:
174 result = nesl_video_read_port(&g_bus.subsystem.video, address);
175 break;
176 case 0x4015:
177 result = nesl_audio_read(&g_bus.subsystem.audio, address);
178 break;
179 case 0x4016 ... 0x4017:
180 result = nesl_input_read(&g_bus.subsystem.input, address);
181 break;
182 case 0x6000 ... 0x7FFF:
183 result = nesl_mapper_read(&g_bus.subsystem.mapper, BANK_PROGRAM_RAM, address);
184 break;
185 case 0x8000 ... 0xFFFF:
186 result = nesl_mapper_read(&g_bus.subsystem.mapper, BANK_PROGRAM_ROM, address);
187 break;
188 default:
189 break;
190 }
191 break;
192 case BUS_VIDEO:
193
194 switch(address) {
195 case 0x0000 ... 0x1FFF:
196 result = nesl_mapper_read(&g_bus.subsystem.mapper, BANK_CHARACTER_ROM, address);
197 break;
198 case 0x2000 ... 0x3FFF:
199 result = nesl_video_read(&g_bus.subsystem.video, address);
200 break;
201 default:
202 break;
203 }
204 break;
205 case BUS_VIDEO_OAM:
206
207 switch(address) {
208 case 0x0000 ... 0x00FF:
209 result = nesl_video_read_oam(&g_bus.subsystem.video, address);
210 break;
211 default:
212 break;
213 }
214 break;
215 default:
216 break;
217 }
218
219 return result;
220}
