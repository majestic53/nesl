88void nesl_audio_cycle(nesl_audio_t *audio, uint64_t cycle)
89{
90
91 if(!(cycle % 6)) {
92 float elapsed = (cycle / (float)89342) / (float)60;
93
94 nesl_audio_square_cycle(&audio->synthesizer.square[SYNTHESIZER_SQUARE_1], elapsed);
95 nesl_audio_square_cycle(&audio->synthesizer.square[SYNTHESIZER_SQUARE_2], elapsed);
96 nesl_audio_triangle_cycle(&audio->synthesizer.triangle, elapsed);
97 nesl_audio_noise_cycle(&audio->synthesizer.noise, elapsed);
98 nesl_audio_dmc_cycle(&audio->synthesizer.dmc, elapsed);
99
100 if((nesl_audio_square_readable(&audio->synthesizer.square[SYNTHESIZER_SQUARE_1]) >= 128)
101 || (nesl_audio_square_readable(&audio->synthesizer.square[SYNTHESIZER_SQUARE_2]) >= 128)
102 || (nesl_audio_triangle_readable(&audio->synthesizer.triangle) >= 128)
103 || (nesl_audio_noise_readable(&audio->synthesizer.noise) >= 128)
104 || (nesl_audio_dmc_readable(&audio->synthesizer.dmc) >= 128)) {
105 float data[SYNTHESIZER_MAX + 1][128] = {};
106
107 nesl_audio_square_read(&audio->synthesizer.square[SYNTHESIZER_SQUARE_1], data[SYNTHESIZER_SQUARE_1], 128);
108 nesl_audio_square_read(&audio->synthesizer.square[SYNTHESIZER_SQUARE_2], data[SYNTHESIZER_SQUARE_2], 128);
109 nesl_audio_triangle_read(&audio->synthesizer.triangle, data[SYNTHESIZER_TRIANGLE], 128);
110 nesl_audio_noise_read(&audio->synthesizer.noise, data[SYNTHESIZER_NOISE], 128);
111 nesl_audio_dmc_read(&audio->synthesizer.dmc, data[SYNTHESIZER_DMC], 128);
112
113 for(int sample = 0; sample < 128; ++sample) {
114
115 for(nesl_synthesizer_e channel = SYNTHESIZER_SQUARE_1; channel < SYNTHESIZER_MAX; ++channel) {
116
117 if(data[channel][sample]) {
118
119 /* TODO: ASSIGN CHANNEL VOLUME */
120 data[SYNTHESIZER_MAX][sample] += data[channel][sample] * 1.f;
121 /* --- */
122 }
123 }
124
125 /* TODO: ASSIGN MASTER VOLUME */
126 data[SYNTHESIZER_MAX][sample] *= 1.f;
127 /* --- */
128
129 if(data[SYNTHESIZER_MAX][sample] < -1.f) {
130 data[SYNTHESIZER_MAX][sample] = -1.f;
131 } else if(data[SYNTHESIZER_MAX][sample] > 1.f) {
132 data[SYNTHESIZER_MAX][sample] = 1.f;
133 }
134 }
135
136 nesl_audio_buffer_write(&audio->buffer, data[SYNTHESIZER_MAX], 128);
137 }
138
139
140 }

143nesl_error_e nesl_audio_initialize(nesl_audio_t *audio)
144{
145 nesl_error_e result;
146
147 if((result = nesl_audio_buffer_initialize(&audio->buffer, 1024)) == NESL_FAILURE) {
148 goto exit;
149 }
150
151 for(nesl_synthesizer_e channel = SYNTHESIZER_SQUARE_1; channel <= SYNTHESIZER_SQUARE_2; ++channel) {
152
153 if((result = nesl_audio_square_initialize(&audio->synthesizer.square[channel])) == NESL_FAILURE) {
154 goto exit;
155 }
156 }
157
158 if((result = nesl_audio_triangle_initialize(&audio->synthesizer.triangle)) == NESL_FAILURE) {
159 goto exit;
160 }
161
162 if((result = nesl_audio_noise_initialize(&audio->synthesizer.noise)) == NESL_FAILURE) {
163 goto exit;
164 }
165
166 if((result = nesl_audio_dmc_initialize(&audio->synthesizer.dmc)) == NESL_FAILURE) {
167 goto exit;
168 }
169
170 if((result = nesl_audio_reset(audio)) == NESL_FAILURE) {
171 goto exit;
172 }
173
174exit:
175 return result;
176}

245void nesl_audio_write(nesl_audio_t *audio, uint16_t address, uint8_t data)
246{
247 nesl_synthesizer_e channel;
248
249 switch(address) {
250 case 0x4000 ... 0x4007:
251
252 if((channel = (address > 0x4003) ? SYNTHESIZER_SQUARE_2 : SYNTHESIZER_SQUARE_1) == SYNTHESIZER_SQUARE_2) {
253 address -= 4;
254 }
255
256 nesl_audio_square_write(&audio->synthesizer.square[channel], address, data);
257 break;
258 case 0x4008 ... 0x400B:
259 nesl_audio_triangle_write(&audio->synthesizer.triangle, address, data);
260 break;
261 case 0x400C ... 0x400F:
262 nesl_audio_noise_write(&audio->synthesizer.noise, address, data);
263 break;
264 case 0x4010 ... 0x4013:
265 nesl_audio_dmc_write(&audio->synthesizer.dmc, address, data);
266 break;
267 case 0x4015:
268 nesl_audio_set_status(audio, data);
269 break;
270 case 0x4017:
271 nesl_audio_set_frame(audio, data);
272 break;
273 default:
274 break;
275 }
276}
