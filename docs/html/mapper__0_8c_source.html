33nesl_error_e nesl_mapper_0_initialize(nesl_mapper_t *mapper)
34{
35 mapper->extension.interrupt = &nesl_mapper_0_interrupt;
36 mapper->extension.read_ram = &nesl_mapper_0_read_ram;
37 mapper->extension.read_rom = &nesl_mapper_0_read_rom;
38 mapper->extension.reset = &nesl_mapper_0_reset;
39 mapper->extension.write_ram = &nesl_mapper_0_write_ram;
40 mapper->extension.write_rom = &nesl_mapper_0_write_rom;
41 mapper->ram.program = 0;
42 mapper->rom.character[0] = 0;
43 mapper->rom.program[0] = 0;
44 mapper->rom.program[1] = (nesl_cartridge_get_banks(&mapper->cartridge, BANK_PROGRAM_ROM) > 1) ? (16 * 1024) : 0;
45
46 return NESL_SUCCESS;
47}

54uint8_t nesl_mapper_0_read_ram(nesl_mapper_t *mapper, nesl_bank_e type, uint16_t address)
55{
56 uint8_t result = 0;
57
58 switch(type) {
59 case BANK_PROGRAM_RAM:
60
61 switch(address) {
62 case 0x6000 ... 0x7FFF:
63 result = nesl_cartridge_read_ram(&mapper->cartridge, BANK_PROGRAM_RAM, mapper->ram.program + (address & 0x1FFF));
64 break;
65 default:
66 break;
67 }
68 break;
69 default:
70 break;
71 }
72
73 return result;
74}

76uint8_t nesl_mapper_0_read_rom(nesl_mapper_t *mapper, nesl_bank_e type, uint16_t address)
77{
78 uint8_t result = 0;
79
80 switch(type) {
81 case BANK_CHARACTER_ROM:
82
83 switch(address) {
84 case 0x0000 ... 0x1FFF:
85 result = nesl_cartridge_read_rom(&mapper->cartridge, BANK_CHARACTER_ROM, mapper->rom.character[0] + (address & 0x1FFF));
86 break;
87 default:
88 break;
89 }
90 break;
91 case BANK_PROGRAM_ROM:
92
93 switch(address) {
94 case 0x8000 ... 0xBFFF:
95 result = nesl_cartridge_read_rom(&mapper->cartridge, BANK_PROGRAM_ROM, mapper->rom.program[0] + (address & 0x3FFF));
96 break;
97 case 0xC000 ... 0xFFFF:
98 result = nesl_cartridge_read_rom(&mapper->cartridge, BANK_PROGRAM_ROM, mapper->rom.program[1] + (address & 0x3FFF));
99 break;
100 default:
101 break;
102 }
103 break;
104 default:
105 break;
106 }
107
108 return result;
109}
