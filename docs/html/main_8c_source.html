68static nesl_error_e read_file(nesl_t *input, char *base, char *path)
69{
70 FILE *file = NULL;
71 nesl_error_e result = NESL_SUCCESS;
72
73 if(!(file = fopen(path, "rb"))) {
74 TRACE(NESL_FAILURE, "%s: File does not exist -- %s\n", base, path);
75 result = NESL_FAILURE;
76 goto exit;
77 }
78
79 fseek(file, 0, SEEK_END);
80
81 if((input->length = ftell(file)) <= 0) {
82 TRACE(NESL_FAILURE, "%s: File is empty -- %s\n", base, path);
83 result = NESL_FAILURE;
84 goto exit;
85 }
86
87 fseek(file, 0, SEEK_SET);
88
89 if(!(input->data = calloc(input->length, sizeof(uint8_t)))) {
90 TRACE(NESL_FAILURE, "%s: Failed to allocate buffer -- %.2f KB (%i bytes)\n", base, input->length / 1024.f, input->length);
91 result = NESL_FAILURE;
92 goto exit;
93 }
94
95 if(fread(input->data, sizeof(uint8_t), input->length, file) != input->length) {
96 TRACE(NESL_FAILURE, "%s: Failed to read file -- %s\n", base, path);
97 result = NESL_FAILURE;
98 goto exit;
99 }
100
101 input->title = basename(path);
102
103exit:
104
105 if(file) {
106 fclose(file);
107 file = NULL;
108 }
109
110 return result;
111}

139{
140

161int main(int argc, char *argv[])
162{
163 int option;
164 nesl_t input = {};
165 nesl_error_e result = NESL_SUCCESS;
166
167 opterr = 1;
168
169 while((option = getopt(argc, argv, "fhls:v")) != -1) {
170
171 switch(option) {
172 case 'f':
173 input.fullscreen = true;
174 break;
175 case 'h':
176 show_help(stdout, true);
177 goto exit;
178 case 'l':
179 input.linear = true;
180 break;
181 case 's':
182 input.scale = strtol(optarg, NULL, 10);
183 break;
184 case 'v':
185 show_version(stdout, false);
186 goto exit;
187 case '?':
188 default:
189 result = NESL_FAILURE;
190 goto exit;
191 }
192 }
193
194 for(option = optind; option < argc; ++option) {
195
196 if((result = read_file(&input, argv[0], argv[option])) == NESL_FAILURE) {
197 goto exit;
198 }
199 break;
200 }
201
202 if((result = nesl(&input)) == NESL_FAILURE) {
203 TRACE(NESL_FAILURE, "%s: %s\n", argv[0], nesl_get_error());
204 goto exit;
205 }
206
207exit:
208
209 if(input.data) {
210 free(input.data);
211 input.data = NULL;
212 }
213
214 return (int)result;
215}
