52static void nesl_video_background_load(nesl_video_t *video)
53{
54
55 switch((video->cycle - 1) % 8) {
56 case 0:
57 video->background.attribute.lsb.low = (video->background.attribute.data & 1) ? 0xFF : 0;
58 video->background.attribute.msb.low = (video->background.attribute.data & 2) ? 0xFF : 0;
59 video->background.pattern.lsb.low = video->background.pattern.data.low;
60 video->background.pattern.msb.low = video->background.pattern.data.high;
61 video->background.type = nesl_bus_read(BUS_VIDEO, 0x2000 + (video->address.v.word & 0x0FFF));
62 break;
63 case 2:
64 video->background.attribute.data = nesl_bus_read(BUS_VIDEO, 0x23C0 + (video->address.v.nametable_y << 11)
65 + (video->address.v.nametable_x << 10) + ((video->address.v.coarse_y >> 2) << 3) + (video->address.v.coarse_x >> 2));
66
67 if(video->address.v.coarse_y & 2) {
68 video->background.attribute.data >>= 4;
69 }
70
71 if(video->address.v.coarse_x & 2) {
72 video->background.attribute.data >>= 2;
73 }
74
75 video->background.attribute.data &= 3;
76 break;
77 case 4:
78 video->background.pattern.data.low = nesl_bus_read(BUS_VIDEO, (video->port.control.background_pattern << 12)
79 + ((uint16_t)video->background.type << 4) + video->address.v.fine_y);
80 break;
81 case 6:
82 video->background.pattern.data.high = nesl_bus_read(BUS_VIDEO, (video->port.control.background_pattern << 12)
83 + ((uint16_t)video->background.type << 4) + video->address.v.fine_y + 8);
84 break;
85 case 7:
86
87 if(video->port.mask.background_show || video->port.mask.sprite_show) {
88
89 if(video->address.v.coarse_x == 31) {
90 video->address.v.coarse_x = 0;
91 video->address.v.nametable_x = ~video->address.v.nametable_x;
92 } else {
93 ++video->address.v.coarse_x;
94 }
95 }
96 break;
97 default:
98 break;
99 }
100}

122static uint8_t nesl_video_flip(uint8_t data)
123{
124 data = ((data & 0x0F) << 4) | ((data & 0xF0) >> 4);
125 data = ((data & 0x33) << 2) | ((data & 0xCC) >> 2);
126 data = ((data & 0x55) << 1) | ((data & 0xAA) >> 1);
127
128 return data;
129}

226static uint16_t nesl_video_nametable_address(uint16_t address, nesl_mirror_e mirror, int *bank)
227{
228
229 switch(mirror) {
230 case MIRROR_HORIZONTAL:
231
232 switch(address) {
233 case 0x2000 ... 0x23FF:
234 case 0x2400 ... 0x27FF:
235 *bank = 0;
236 break;
237 case 0x2800 ... 0x2BFF:
238 case 0x2C00 ... 0x2FFF:
239 *bank = 1;
240 break;
241 default:
242 break;
243 }
244 break;
245 case MIRROR_VERTICAL:
246
247 switch(address) {
248 case 0x2000 ... 0x23FF:
249 case 0x2800 ... 0x2BFF:
250 *bank = 0;
251 break;
252 case 0x2400 ... 0x27FF:
253 case 0x2C00 ... 0x2FFF:
254 *bank = 1;
255 break;
256 default:
257 break;
258 }
259 break;
260 case MIRROR_ONE_LOW:
261 *bank = 0;
262 break;
263 case MIRROR_ONE_HIGH:
264 *bank = 1;
265 break;
266 default:
267 break;
268 }
269
270 return address & 0x03FF;
271}

299static void nesl_video_render(nesl_video_t *video)
300{
301 bool priority;
302 uint8_t color[3] = {}, palette[3] = {};
303
304 if(video->port.mask.background_show) {
305 uint16_t mask = 0x8000 >> video->address.fine_x;
306
307 color[1] = (((video->background.pattern.msb.word & mask) > 0) << 1) | ((video->background.pattern.lsb.word & mask) > 0);
308 palette[1] = (((video->background.attribute.msb.word & mask) > 0) << 1) | ((video->background.attribute.lsb.word & mask) > 0);
309 }
310
311 if(video->port.mask.sprite_show) {
312 video->sprite.sprite_0_render = false;
313
314 for(uint8_t index = 0; index < video->sprite.count; ++index) {
315 const nesl_video_object_t *object = &video->sprite.object[index];
316
317 if(!object->x) {
318 color[2] = (((video->sprite.pattern[index].msb & 0x80) > 0) << 1) | ((video->sprite.pattern[index].lsb & 0x80) > 0);
319 palette[2] = object->attribute.palette + 4;
320 priority = (object->attribute.priority == 0);
321
322 if(color[2]) {
323
324 if(!index) {
325 video->sprite.sprite_0_render = true;
326 }
327
328 break;
329 }
330 }
331 }
332 }
333
334 if(!color[1] && !color[2]) {
335 color[0] = 0;
336 palette[0] = 0;
337 } else if(color[1] && !color[2]) {
338 color[0] = color[1];
339 palette[0] = palette[1];
340 } else if(!color[1] && color[2]) {
341 color[0] = color[2];
342 palette[0] = palette[2];
343 } else {
344 color[0] = color[priority ? 2 : 1];
345 palette[0] = palette[priority ? 2 : 1];
346
347 if(video->sprite.sprite_0_found && video->sprite.sprite_0_render
348 && video->port.mask.background_show && video->port.mask.sprite_show) {
349
350 if(video->port.mask.background_left_show || video->port.mask.sprite_left_show) {
351
352 if((video->cycle > 0) && (video->cycle <= 257)) {
353 video->port.status.sprite_0_hit = true;
354 }
355 } else if((video->cycle > 8) && (video->cycle <= 257)) {
356 video->port.status.sprite_0_hit = true;
357 }
358 }
359 }
360
361 if((video->scanline >= 0) && (video->scanline < 240) && ((video->cycle - 1) < 256)) {
362 nesl_service_set_pixel(nesl_bus_read(BUS_VIDEO, 0x3F00 + (4 * palette[0]) + color[0]),
363 video->port.mask.red_emphasis, video->port.mask.green_emphasis, video->port.mask.blue_emphasis,
364 video->cycle - 1, video->scanline);
365 }
366}

476static void nesl_video_sprite_evaluate(nesl_video_t *video)
477{
478 video->port.status.sprite_overflow = false;
479 memset(&video->sprite.object, 0xFF, sizeof(video->sprite.object));
480 video->sprite.sprite_0_found = false;
481 video->sprite.count = 0;
482
483 for(uint8_t index = 0; index < 64; ++index) {
484 const nesl_video_object_t *object = &video->ram.oam[index];
485 int16_t offset = ((int16_t)video->scanline - (int16_t)object->y);
486
487 if((offset >= 0) && (offset < (video->port.control.sprite_size ? 16 : 8))) {
488
489 if(video->sprite.count < 8) {
490 video->sprite.object[video->sprite.count++].raw = object->raw;
491
492 if(!index) {
493 video->sprite.sprite_0_found = true;
494 }
495 } else {
496 video->port.status.sprite_overflow = true;
497 break;
498 }
499 }
500 }
501}

507static void nesl_video_sprite_load(nesl_video_t *video)
508{
509
510 for(uint8_t index = 0; index < video->sprite.count; ++index) {
511 nesl_register_t address = {};
512 const nesl_video_object_t *object = &video->sprite.object[index];
513
514 if(video->port.control.sprite_size) {
515
516 if(object->attribute.flip_vertical) {
517
518 if((video->scanline - object->y) < 8) {
519 address.word = (object->type.bank << 12) + (((object->type.raw & 0xFE) + 1) << 4) + ((7 - (video->scanline - object->y)) & 7);
520 } else {
521 address.word = (object->type.bank << 12) + ((object->type.raw & 0xFE) << 4) + ((7 - (video->scanline - object->y)) & 7);
522 }
523 } else {
524
525 if((video->scanline - object->y) < 8) {
526 address.word = (object->type.bank << 12) + ((object->type.raw & 0xFE) << 4) + ((video->scanline - object->y) & 7);
527 } else {
528 address.word = (object->type.bank << 12) + (((object->type.raw & 0xFE) + 1) << 4) + ((video->scanline - object->y) & 7);
529 }
530 }
531 } else {
532
533 if(object->attribute.flip_vertical) {
534 address.word = (video->port.control.sprite_pattern << 12) + (object->type.raw << 4) + (7 - (video->scanline - object->y));
535 } else {
536 address.word = (video->port.control.sprite_pattern << 12) + (object->type.raw << 4) + (video->scanline - object->y);
537 }
538 }
539
540 video->sprite.pattern[index].lsb = nesl_bus_read(BUS_VIDEO, address.word);
541 video->sprite.pattern[index].msb = nesl_bus_read(BUS_VIDEO, address.word + 8);
542
543 if(object->attribute.flip_horizontal) {
544 video->sprite.pattern[index].lsb = nesl_video_flip(video->sprite.pattern[index].lsb);
545 video->sprite.pattern[index].msb = nesl_video_flip(video->sprite.pattern[index].msb);
546 }
547 }
548}

644bool nesl_video_cycle(nesl_video_t *video)
645{
646 bool result = false;
647
648 switch(video->scanline) {
649 case (-1):
650
651 if(video->cycle == 1) {
652 nesl_video_vertical_blank_exit(video);
653 }
654 case 0 ... 239:
655
656 switch(video->cycle) {
657 case 1 ... 256:
658 case 321 ... 337:
659
660 if(video->cycle > 1) {
661 nesl_video_background_shift(video);
662
663 if(video->cycle <= 256) {
664 nesl_video_sprite_shift(video);
665 }
666 }
667
668 nesl_video_background_load(video);
669
670 if(video->cycle == 256) {
671 nesl_video_vertical_increment(video);
672 }
673 break;
674 case 257:
675 nesl_video_horizontal_set(video);
676
677 if(video->scanline >= 0) {
678 nesl_video_sprite_evaluate(video);
679 nesl_video_sprite_load(video);
680 }
681 break;
682 case 260:
683 nesl_video_mapper_interrupt(video);
684 break;
685 case 280 ... 304:
686
687 if(video->scanline == -1) {
688 nesl_video_vertical_set(video);
689 }
690 break;
691 default:
692 break;
693 }
694 break;
695 case 241:
696
697 if(video->cycle == 1) {
698 nesl_video_vertical_blank(video);
699 }
700 break;
701 default:
702 break;
703 }
704
705 nesl_video_render(video);
706
707 if(++video->cycle > 340) {
708 video->cycle = 0;
709
710 if((result = (++video->scanline > 260))) {
711 video->scanline = (-1);
712 }
713 }
714
715 return result;
716}

731{

744 break;

746 break;
