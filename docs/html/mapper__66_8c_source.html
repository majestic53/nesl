37static void nesl_mapper_66_set(nesl_mapper_t *mapper)
38{
39 mapper->rom.character[0] = ((nesl_mapper_66_t *)mapper->context)->bank.character * 8 * 1024;
40 mapper->rom.program[0] = ((nesl_mapper_66_t *)mapper->context)->bank.program * 32 * 1024;
41}

43nesl_error_e nesl_mapper_66_initialize(nesl_mapper_t *mapper)
44{
45 nesl_error_e result = NESL_SUCCESS;
46
47 if(!(mapper->context = calloc(1, sizeof(nesl_mapper_66_t)))) {
48 result = SET_ERROR("Failed to allocate buffer -- %u KB (%i bytes)", sizeof(nesl_mapper_66_t), sizeof(nesl_mapper_66_t));
49 goto exit;
50 }
51
52 mapper->extension.interrupt = &nesl_mapper_66_interrupt;
53 mapper->extension.read_ram = &nesl_mapper_66_read_ram;
54 mapper->extension.read_rom = &nesl_mapper_66_read_rom;
55 mapper->extension.reset = &nesl_mapper_66_reset;
56 mapper->extension.write_ram = &nesl_mapper_66_write_ram;
57 mapper->extension.write_rom = &nesl_mapper_66_write_rom;
58
59 if((result = nesl_mapper_66_reset(mapper)) == NESL_FAILURE) {
60 goto exit;
61 }
62
63exit:
64 return result;
65}

72uint8_t nesl_mapper_66_read_ram(nesl_mapper_t *mapper, nesl_bank_e type, uint16_t address)
73{
74 uint8_t result = 0;
75
76 switch(type) {
77 case BANK_PROGRAM_RAM:
78
79 switch(address) {
80 case 0x6000 ... 0x7FFF:
81 result = nesl_cartridge_read_ram(&mapper->cartridge, BANK_PROGRAM_RAM, mapper->ram.program + (address & 0x1FFF));
82 break;
83 default:
84 break;
85 }
86 break;
87 default:
88 break;
89 }
90
91 return result;
92}

164{

172 break;
