39static void nesl_audio_buffer_copy_in(nesl_audio_buffer_t *buffer, float *data, int length)
40{
41
42 if((buffer->write + length) >= buffer->length) {
43 int offset = buffer->length - buffer->write;
44
45 memcpy(&buffer->data[buffer->write], data, offset * sizeof(*data));
46 length -= offset;
47 data += offset;
48 buffer->write = 0;
49 }
50
51 memcpy(&buffer->data[buffer->write], data, length * sizeof(*data));
52 buffer->write += length;
53 buffer->full = (buffer->write == buffer->read);
54}

62static void nesl_audio_buffer_copy_out(nesl_audio_buffer_t *buffer, float *data, int length)

64

71 buffer->read = 0;

74 memcpy(data, &buffer->data[buffer->read], length * sizeof(*data));
75 buffer->read += length;
76 buffer->full = false;

86static int nesl_audio_buffer_distance(int max, int left, int right)
87{
88 int result = 0;
89
90 if(left <= right) {
91 result = right - left;
92 } else {
93 result = (max - left) + right;
94 }
95
96 return result;

128nesl_error_e nesl_audio_buffer_initialize(nesl_audio_buffer_t *buffer, int length)
129{
130 nesl_error_e result = NESL_SUCCESS;
131
132 if(pthread_mutex_init(&buffer->lock, NULL) == -1) {
133 result = SET_ERROR("Failed to initialize lock -- %i: %s", errno, strerror(errno));
134 goto exit;
135 }
136
137 if(!(buffer->data = calloc(length, sizeof(*buffer->data)))) {
138 result = SET_ERROR("Failed to allocate buffer -- %u KB (%i bytes)", length / 1024.f, length);
139 goto exit;
140 }
141
142 buffer->length = length;
143
144 if((result = nesl_audio_buffer_reset(buffer)) == NESL_FAILURE) {
145 goto exit;
146 }
147
148exit:
149 return result;
150}

152int nesl_audio_buffer_read(nesl_audio_buffer_t *buffer, float *data, int length)
153{
154 int result = 0;
155
156 pthread_mutex_lock(&buffer->lock);
157
158 if(!nesl_audio_buffer_is_empty(buffer)) {
159
160 if((result = nesl_audio_buffer_minimum(nesl_audio_buffer_distance(buffer->length, buffer->read, buffer->write), length)) > 0) {
161 nesl_audio_buffer_copy_out(buffer, data, result);
162 }
163 }
164
165 pthread_mutex_unlock(&buffer->lock);
166
167 return result;
168}

207int nesl_audio_buffer_write(nesl_audio_buffer_t *buffer, float *data, int length)
208{
209 int result = 0;
210
211 pthread_mutex_lock(&buffer->lock);
212
213 if(!nesl_audio_buffer_is_full(buffer)) {
214
215 if(!(result = nesl_audio_buffer_distance(buffer->length, buffer->write, buffer->read))) {
216 result = buffer->length;
217 }
218
219 if((result = nesl_audio_buffer_minimum(result, length)) > 0) {
220 nesl_audio_buffer_copy_in(buffer, data, result);
221 }
222 }
223
224 pthread_mutex_unlock(&buffer->lock);
225
226 return result;
227}
