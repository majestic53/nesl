133nesl_error_e nesl_service_initialize(const char *title, int fullscreen, int linear, int scale)
134{
135 nesl_error_e result = NESL_SUCCESS;
136
137 if(scale < 1) {
138 scale = 1;
139 } else if(scale > 4) {
140 scale = 4;
141 }
142
143 if(SDL_Init(SDL_INIT_AUDIO | SDL_INIT_VIDEO)) {
144 result = SET_ERROR("%s", SDL_GetError());
145 goto exit;
146 }
147
148 if(!(g_service.handle.window = SDL_CreateWindow(title, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, 256 * scale, 240 * scale, SDL_WINDOW_RESIZABLE))) {
149 result = SET_ERROR("%s", SDL_GetError());
150 goto exit;
151 }
152
153 if(!(g_service.handle.renderer = SDL_CreateRenderer(g_service.handle.window, -1, SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC))) {
154 result = SET_ERROR("%s", SDL_GetError());
155 goto exit;
156 }
157
158 if(SDL_RenderSetLogicalSize(g_service.handle.renderer, 256, 240)) {
159 result = SET_ERROR("%s", SDL_GetError());
160 goto exit;
161 }
162
163 if(SDL_SetRenderDrawColor(g_service.handle.renderer, 0, 0, 0, 0)) {
164 result = SET_ERROR("%s", SDL_GetError());
165 goto exit;
166 }
167
168 if(SDL_SetHint(SDL_HINT_RENDER_VSYNC, "1") == SDL_FALSE) {
169 result = SET_ERROR("%s", SDL_GetError());
170 goto exit;
171 }
172
173 if(SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, linear ? "1" : "0") == SDL_FALSE) {
174 result = SET_ERROR("%s", SDL_GetError());
175 goto exit;
176 }
177
178 if(!(g_service.handle.texture = SDL_CreateTexture(g_service.handle.renderer, SDL_PIXELFORMAT_ARGB8888, SDL_TEXTUREACCESS_STREAMING, 256, 240))) {
179 result = SET_ERROR("%s", SDL_GetError());
180 goto exit;
181 }
182
183 if(fullscreen) {
184
185 if((result = nesl_service_toggle_fullscreen()) == NESL_FAILURE) {
186 goto exit;
187 }
188 }
189
190 if((result = nesl_service_reset()) == NESL_FAILURE) {
191 goto exit;
192 }
193
194exit:
195 return result;
196}

198nesl_error_e nesl_service_poll(void)
199{
200 SDL_Event event;
201 nesl_error_e result = NESL_SUCCESS;
202
203 while(SDL_PollEvent(&event)) {
204
205 switch(event.type) {
206 case SDL_KEYUP:
207
208 if(!event.key.repeat) {
209
210 switch(event.key.keysym.scancode) {
211 case SDL_SCANCODE_F11:
212
213 if((result = nesl_service_toggle_fullscreen()) == NESL_FAILURE) {
214 goto exit;
215 }
216 break;
217 case SDL_SCANCODE_R:
218
219 if((result = nesl_bus_interrupt(INTERRUPT_RESET)) == NESL_FAILURE) {
220 goto exit;
221 }
222 break;
223 default:
224 break;
225 }
226 }
227 break;
228 case SDL_QUIT:
229 result = NESL_QUIT;
230 goto exit;
231 default:
232 break;
233 }
234 }
235
236exit:
237 return result;
238}

240nesl_error_e nesl_service_redraw(void)
241{
242 uint32_t elapsed;
243 nesl_error_e result = NESL_SUCCESS;
244
245 if(SDL_UpdateTexture(g_service.handle.texture, NULL, (uint32_t *)g_service.pixel, 256 * sizeof(uint32_t))) {
246 result = SET_ERROR("%s", SDL_GetError());
247 goto exit;
248 }
249
250 if(SDL_RenderClear(g_service.handle.renderer)) {
251 result = SET_ERROR("%s", SDL_GetError());
252 goto exit;
253 }
254
255 if(SDL_RenderCopy(g_service.handle.renderer, g_service.handle.texture, NULL, NULL)) {
256 result = SET_ERROR("%s", SDL_GetError());
257 goto exit;
258 }
259
260 if((elapsed = (SDL_GetTicks() - g_service.tick)) < (1000 / (float)60)) {
261 SDL_Delay((1000 / (float)60) - elapsed);
262 }
263
264 SDL_RenderPresent(g_service.handle.renderer);
265 g_service.tick = SDL_GetTicks();
266
267exit:
268 return result;
269}

286nesl_error_e nesl_service_set_audio(nesl_service_get_audio callback, void *context)
287{
288 nesl_error_e result = NESL_SUCCESS;
289 SDL_AudioSpec desired = {}, obtained = {};
290
291 desired.callback = callback;
292 desired.channels = 1;
293 desired.format = AUDIO_F32SYS;
294 desired.freq = 44100;
295 desired.samples = 512;
296 desired.userdata = context;
297 nesl_service_close_audio();
298
299 if((g_service.handle.audio = SDL_OpenAudioDevice(NULL, 0, &desired, &obtained, 0)) <= 0) {
300 result = SET_ERROR("%s", SDL_GetError());
301 goto exit;
302 }
303
304 SDL_PauseAudioDevice(g_service.handle.audio, 0);
305
306exit:
307 return result;
308}
