68static nesl_error_e nesl_mapper_extension_initialize(nesl_mapper_t *mapper)
69{
70 nesl_error_e result = NESL_FAILURE;
71 const nesl_mapper_extension_t *extension = NULL;
72 int count = sizeof(CONTEXT) / sizeof(*(CONTEXT)), index;
73
74 for(index = 0; index < count; ++index) {
75 extension = &CONTEXT[index];
76
77 if(extension->type == mapper->type) {
78
79 if((result = extension->initialize(mapper)) == NESL_FAILURE) {
80 goto exit;
81 }
82 break;
83 }
84 }
85
86 if(index == count) {
87 result = SET_ERROR("Unsupported mapper type -- %u", mapper->type);
88 goto exit;
89 }
90
91exit:
92 return result;
93}

99static void nesl_mapper_extension_uninitialize(nesl_mapper_t *mapper)

112nesl_error_e nesl_mapper_initialize(nesl_mapper_t *mapper, const void *data, int length)
113{
114 nesl_error_e result;
115
116 if((result = nesl_cartridge_initialize(&mapper->cartridge, data, length)) == NESL_FAILURE) {
117 goto exit;
118 }
119
120 mapper->mirror = nesl_cartridge_get_mirror(&mapper->cartridge);
121 mapper->type = nesl_cartridge_get_mapper(&mapper->cartridge);
122
123 if((result = nesl_mapper_extension_initialize(mapper)) == NESL_FAILURE) {
124 goto exit;
125 }
126
127 if((result = nesl_mapper_reset(mapper)) == NESL_FAILURE) {
128 goto exit;
129 }
130
131exit:
132 return result;
133}

140uint8_t nesl_mapper_read(nesl_mapper_t *mapper, nesl_bank_e type, uint16_t address)
141{
142 uint8_t result = 0;
143
144 switch(type) {
145 case BANK_CHARACTER_RAM:
146 case BANK_PROGRAM_RAM:
147 result = mapper->extension.read_ram(mapper, type, address);
148 break;
149 case BANK_CHARACTER_ROM:
150 case BANK_PROGRAM_ROM:
151 result = mapper->extension.read_rom(mapper, type, address);
152 break;
153 default:
154 break;
155 }
156
157 return result;
158}
