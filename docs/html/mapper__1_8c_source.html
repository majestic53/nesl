37static void nesl_mapper_1_set_character(nesl_mapper_t *mapper)
38{
39 const nesl_mapper_1_t *context = mapper->context;
40
41 switch(context->control.character) {
42 case 0:
43 mapper->rom.character[0] = (context->character[0].bank >> 1) * 8 * 1024;
44 mapper->rom.character[1] = mapper->rom.character[0] + (4 * 1024);
45 break;
46 case 1:
47 mapper->rom.character[0] = context->character[0].bank * 4 * 1024;
48 mapper->rom.character[1] = context->character[1].bank * 4 * 1024;
49 break;
50 default:
51 break;
52 }
53}

84static void nesl_mapper_1_set_program(nesl_mapper_t *mapper)
85{
86 const nesl_mapper_1_t *context = mapper->context;
87
88 switch(context->control.program) {
89 case 0 ... 1:
90 mapper->rom.program[0] = (context->program.bank >> 1) * 32 * 1024;
91 mapper->rom.program[1] = mapper->rom.program[0] + (16 * 1024);
92 break;
93 case 2:
94 mapper->rom.program[0] = 0;
95 mapper->rom.program[1] = context->program.bank * 16 * 1024;
96 break;
97 case 3:
98 mapper->rom.program[0] = context->program.bank * 16 * 1024;
99 mapper->rom.program[1] = (nesl_cartridge_get_banks(&mapper->cartridge, BANK_PROGRAM_ROM) - 1) * 16 * 1024;
100 break;
101 default:
102 break;
103 }

112static void nesl_mapper_1_set(nesl_mapper_t *mapper, uint16_t address, uint8_t data)
113{
114 nesl_mapper_1_t *context = mapper->context;
115
116 if(data & 0x80) {
117 context->shift.data = 0;
118 context->shift.position = 0;
119 } else {
120 context->shift.data = ((context->shift.data >> 1) | ((data & 1) << 4)) & 0x1F;
121
122 if(++context->shift.position == 5) {
123
124 switch(address) {
125 case 0x8000 ... 0x9FFF:
126 context->control.raw = context->shift.data;
127 break;
128 case 0xA000 ... 0xBFFF:
129 context->character[0].raw = context->shift.data;
130 break;
131 case 0xC000 ... 0xDFFF:
132 context->character[1].raw = context->shift.data;
133 break;
134 case 0xE000 ... 0xFFFF:
135 context->program.raw = context->shift.data;
136 break;
137 default:
138 break;
139 }
140
141 nesl_mapper_1_set_mirror(mapper);
142 nesl_mapper_1_set_character(mapper);
143 nesl_mapper_1_set_program(mapper);
144 context->shift.data = 0;
145 context->shift.position = 0;
146 }
147 }
148}

150nesl_error_e nesl_mapper_1_initialize(nesl_mapper_t *mapper)
151{
152 nesl_error_e result = NESL_SUCCESS;
153
154 if(!(mapper->context = calloc(1, sizeof(nesl_mapper_1_t)))) {
155 result = SET_ERROR("Failed to allocate buffer -- %u KB (%i bytes)", sizeof(nesl_mapper_1_t), sizeof(nesl_mapper_1_t));
156 goto exit;
157 }
158
159 mapper->extension.interrupt = &nesl_mapper_1_interrupt;
160 mapper->extension.read_ram = &nesl_mapper_1_read_ram;
161 mapper->extension.read_rom = &nesl_mapper_1_read_rom;
162 mapper->extension.reset = &nesl_mapper_1_reset;
163 mapper->extension.write_ram = &nesl_mapper_1_write_ram;
164 mapper->extension.write_rom = &nesl_mapper_1_write_rom;
165
166 if((result = nesl_mapper_1_reset(mapper)) == NESL_FAILURE) {
167 goto exit;
168 }
169
170exit:
171 return result;
172}

179uint8_t nesl_mapper_1_read_ram(nesl_mapper_t *mapper, nesl_bank_e type, uint16_t address)
180{
181 uint8_t result = 0;
182
183 switch(type) {
184 case BANK_PROGRAM_RAM:
185
186 switch(address) {
187 case 0x6000 ... 0x7FFF:
188
189 if(!((nesl_mapper_1_t *)mapper->context)->program.ram_disable) {
190 result = nesl_cartridge_read_ram(&mapper->cartridge, BANK_PROGRAM_RAM, mapper->ram.program + (address & 0x1FFF));
191 }
192 break;
193 default:
194 break;
195 }
196 break;
197 default:
198 break;
199 }
200
201 return result;
202}
